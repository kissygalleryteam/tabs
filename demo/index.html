代码 
S.use('kg/tabs/1.0.0/, kg/tabs/1.0.0/plugins/navFixed/navFixed, node, kg/tabs/1.0.0/index.css', function (S, Tabs, Fixed, Node) {
    var $ = Node.all;
    tab1 = new Tabs({
        $target: '#J_Tab1',
        scrollOffset: 10
    });
    tab1.render();
    tab1.plug(new Fixed());
    $('#disable2').on('click', function(){
        tab1.on('switch:before', function(e){
            if (e.index === 1) {
                return false;
            }
        });
    });
    $('#reset2').on('click', function(){
        tab1.detach('switch:before');
    });
});
